function calculateBmi(weight, height) {
            
   if (weight > 0 && height > 0) {
        var finalBmi = (weight / (height * height)).toFixed(2)
        document.write("BMI=" + finalBmi);
        if (finalBmi < 18.5) {
            document.write("That you are too thin.");
        }
        if (finalBmi > 18.5 && finalBmi < 25) {
            document.write("That you are healthy.")
        }
        if (finalBmi > 25) {
            document.write("That you have overweight.");
        }
    }
    else {
        document.write("Please Fill in everything correctly");
    }
}
